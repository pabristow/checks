Header <boost/checks/ean.hpp>

This file provides tools to compute and validate an European Article Numbering (EAN) of size 8 or 13.

namespace boost {
  namespace checks {
    typedef boost::checks::modulus10_algorithm< ean_weight, ean_sense, 0 > ean_check_algorithm;  // This is the type of the EAN algorithm for validating a check digit. 
    typedef boost::checks::modulus10_algorithm< ean_weight, ean_sense, 1 > ean_compute_algorithm;  // This is the type of the EAN algorithm for computing a check digit. 
    typedef boost::checks::rightmost ean_sense;  // This is the running sense or direction to check an EAN. 
    typedef boost::checks::weight< 1, 3 > ean_weight;  // This is the weight used by EAN system. 
  }
}
